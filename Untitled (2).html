# use the prophet forecasting model - https://facebook.github.io/prophet/

m = Prophet()
m.fit(df_train)


forecast = m.predict(df_test)

results = pd.merge(forecast,df_test,on="ds")

score = mean_absolute_percentage_error(df_test["y"], results["yhat"])

print(f"the prophet mape is: {score}")

plt.plot("Date","y","",data=df_test)
plt.plot(results["Date"], results["yhat"])
plt.show()

#better, but still lots of room to improve
